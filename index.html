Comment configurer ?

1. Créez un compte gratuit sur supabase.com
2. Créez un projet → SQL Editor → exécutez le script ci-dessous
3. Allez dans Settings → API
4. Copiez Project URL et anon public key
5. Collez-les ci-dessous et cliquez Connecter

Script SQL — à exécuter une seule fois dans Supabase → SQL Editor
-- ============================================================
-- INVOORENT - Schema + RLS (usage perso / petite equipe)
-- Executer ce script UNE SEULE FOIS dans Supabase SQL Editor
-- ============================================================

create table if not exists autoloc_vehicules (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists autoloc_clients (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists autoloc_reservations (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists autoloc_maintenances (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);

-- Activer RLS sur toutes les tables
alter table autoloc_vehicules enable row level security;
alter table autoloc_clients enable row level security;
alter table autoloc_reservations enable row level security;
alter table autoloc_maintenances enable row level security;

-- Supprimer d'anciennes policies trop larges si elles existent
drop policy if exists "anon all" on autoloc_vehicules;
drop policy if exists "anon all" on autoloc_clients;
drop policy if exists "anon all" on autoloc_reservations;
drop policy if exists "anon all" on autoloc_maintenances;

-- Policies explicites pour role anon (app locale + cle anon)
-- Vehicules
create policy "veh_select_anon" on autoloc_vehicules for select to anon using (true);
create policy "veh_insert_anon" on autoloc_vehicules for insert to anon with check (true);
create policy "veh_update_anon" on autoloc_vehicules for update to anon using (true) with check (true);
create policy "veh_delete_anon" on autoloc_vehicules for delete to anon using (true);

-- Clients
create policy "cl_select_anon" on autoloc_clients for select to anon using (true);
create policy "cl_insert_anon" on autoloc_clients for insert to anon with check (true);
create policy "cl_update_anon" on autoloc_clients for update to anon using (true) with check (true);
create policy "cl_delete_anon" on autoloc_clients for delete to anon using (true);

-- Reservations
create policy "res_select_anon" on autoloc_reservations for select to anon using (true);
create policy "res_insert_anon" on autoloc_reservations for insert to anon with check (true);
create policy "res_update_anon" on autoloc_reservations for update to anon using (true) with check (true);
create policy "res_delete_anon" on autoloc_reservations for delete to anon using (true);

-- Maintenances
create policy "maint_select_anon" on autoloc_maintenances for select to anon using (true);
create policy "maint_insert_anon" on autoloc_maintenances for insert to anon with check (true);
create policy "maint_update_anon" on autoloc_maintenances for update to anon using (true) with check (true);
create policy "maint_delete_anon" on autoloc_maintenances for delete to anon using (true);

-- Notes securite:
-- 1) Utiliser uniquement la cle anon cote client.
-- 2) Ne jamais exposer la cle service_role dans l'app.
-- 3) Pour un usage multi-utilisateur, ajouter owner_id et remplacer
--    (true) par auth.uid() = owner_id.

Utilisez la clé anon public — jamais la clé service_role
