Comment configurer ?

1. Créez un compte gratuit sur supabase.com
2. Créez un projet → SQL Editor → exécutez le script ci-dessous
3. Allez dans Settings → API
4. Copiez Project URL et anon public key
5. Collez-les ci-dessous et cliquez Connecter

Confidentialité des données

La synchronisation Supabase envoie les données en clair dans la colonne data (jsonb). Pour un usage personnel sur un projet privé, cela reste acceptable. Notez toutefois qu'un accès admin Supabase peut lire les données clients, réservations, véhicules et maintenances.

-- ============================================================
-- INVOORENT — schéma + RLS (usage perso / petite équipe)
-- À exécuter UNE SEULE FOIS dans Supabase → SQL Editor
-- Tables : préfixe invoo_ (produit INVOORENT). Si vous aviez déjà
-- autoloc_* d’une ancienne version, renommez-les ou migrez les données.
-- ============================================================

create table if not exists invoo_vehicules (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists invoo_clients (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists invoo_reservations (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);
create table if not exists invoo_maintenances (
 id text primary key,
 data jsonb,
 deleted_at timestamptz,
 updated_at timestamptz default now()
);

-- Activer RLS sur toutes les tables
alter table invoo_vehicules enable row level security;
alter table invoo_clients enable row level security;
alter table invoo_reservations enable row level security;
alter table invoo_maintenances enable row level security;

-- Supprimer d'anciennes policies trop larges si elles existent
drop policy if exists "anon all" on invoo_vehicules;
drop policy if exists "anon all" on invoo_clients;
drop policy if exists "anon all" on invoo_reservations;
drop policy if exists "anon all" on invoo_maintenances;

-- Policies explicites pour role anon (app locale + cle anon)
-- Vehicules
create policy "veh_select_anon" on invoo_vehicules for select to anon using (true);
create policy "veh_insert_anon" on invoo_vehicules for insert to anon with check (true);
create policy "veh_update_anon" on invoo_vehicules for update to anon using (true) with check (true);
create policy "veh_delete_anon" on invoo_vehicules for delete to anon using (true);

-- Clients
create policy "cl_select_anon" on invoo_clients for select to anon using (true);
create policy "cl_insert_anon" on invoo_clients for insert to anon with check (true);
create policy "cl_update_anon" on invoo_clients for update to anon using (true) with check (true);
create policy "cl_delete_anon" on invoo_clients for delete to anon using (true);

-- Reservations
create policy "res_select_anon" on invoo_reservations for select to anon using (true);
create policy "res_insert_anon" on invoo_reservations for insert to anon with check (true);
create policy "res_update_anon" on invoo_reservations for update to anon using (true) with check (true);
create policy "res_delete_anon" on invoo_reservations for delete to anon using (true);

-- Maintenances
create policy "maint_select_anon" on invoo_maintenances for select to anon using (true);
create policy "maint_insert_anon" on invoo_maintenances for insert to anon with check (true);
create policy "maint_update_anon" on invoo_maintenances for update to anon using (true) with check (true);
create policy "maint_delete_anon" on invoo_maintenances for delete to anon using (true);

-- Notes securite:
-- 1) Utiliser uniquement la cle anon cote client.
-- 2) Ne jamais exposer la cle service_role dans l'app.
-- 3) Pour un usage multi-utilisateur, ajouter owner_id et remplacer
--    (true) par auth.uid() = owner_id.
--
-- --- Temps réel (obligatoire pour voir les changements des autres sans F5) ---
-- Sans publication Realtime, la sync REST (toutes les 5 min ou clic) reste seule active.

Utilisez la clé anon public — jamais la clé service_role
